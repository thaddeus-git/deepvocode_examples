How to use app.config.from_envvar? (Flask): 2 Methods + Complete Guide

Quick Answer

Flask’s app.config.from_envvar() loads configuration from a file path specified by an environment variable. Set an environment variable pointing to your config file, then call the method to load those settings into your Flask app.

# Set environment variable: export FLASK_CONFIG=/path/to/config.py
app.config.from_envvar('FLASK_CONFIG')

Choose Your Method

Based on your specific needs:

  • 🚀 Speed Seeker → Method 1: Direct file loading for quick development
  • 📚 Learning Explorer → Method 2: Environment variable approach with detailed explanations
  • 🔧 Problem Solver → Method 1: Straightforward implementation that just works
  • 🏗️ Architecture Builder → Method 3: Conditional loading for scalable applications
  • 🎨 Output Focused → Method 2: Clean separation of config and code
  • ⚡ Legacy Maintainer → Method 1: Compatible with older Flask versions

Method 1: Direct File Loading (Recommended)

Rating: ⭐ Recommended for Development

How it works:

This method directly loads configuration from a file specified by an environment variable. Flask reads the environment variable value as a file path, then executes that Python file and imports its variables into the app configuration.

Complete Implementation:

import os
from flask import Flask
 
def setup_flask_app():
    """Set up Flask application with environment-based configuration."""
    app = Flask(__name__)
    
    # Method 1: Direct loading with error handling
    config_path = os.environ.get('FLASK_CONFIG')
    
    if config_path and os.path.exists(config_path):
        # Load configuration from the specified file
        app.config.from_envvar('FLASK_CONFIG')
        print(f"Configuration loaded from: {config_path}")
    else:
        # Fallback to default configuration
        print("Using default configuration")
        app.config.update(
            DEBUG=True,
            SECRET_KEY='development-key-change-me',
            DATABASE_URL='sqlite:///development.db'
        )
    
    # Validate required configuration
    required_keys = ['SECRET_KEY']
    for key in required_keys:
        if not app.config.get(key):
            raise ValueError(f"Required configuration key '{key}' is missing")
    
    return app
 
# Create application instance
app = setup_flask_app()
 
# Example route to verify configuration
@app.route('/config')
def show_config():
    """Display current configuration (excluding sensitive data)."""
    safe_config = {
        key: value for key, value in app.config.items()
        if 'SECRET' not in key and 'PASSWORD' not in key
    }
    return safe_config
 
if __name__ == '__main__':
    app.run(debug=app.config.get('DEBUG', False))

Pros:

  • Simple and straightforward implementation
  • Direct file path control
  • Easy to debug and troubleshoot
  • Works well for development environments

Cons:

  • Less flexible for different deployment scenarios
  • File path dependencies can be fragile

When to use:

  • Development and testing environments
  • Simple applications with minimal configuration needs
  • When you have direct control over the deployment environment

Method 2: Environment Variable Loading

Rating: ⭐⭐ Best for Production

How it works:

This method uses environment variables to specify configuration file locations, making it ideal for containerized deployments and production environments where configuration needs to be externalized.

Complete Implementation:

import os
from flask import Flask
from pathlib import Path
 
class ConfigurationManager:
    """Manages Flask configuration loading from environment variables."""
    
    def __init__(self, app=None):
        self.app = app
        if app:
            self.init_app(app)
    
    def init_app(self, app):
        """Initialize configuration for Flask application."""
        self.app = app
        self.load_configuration()
    
    def load_configuration(self):
        """Load configuration using environment variable approach."""
        # Set default configuration first
        self._set_defaults()
        
        # Try to load from environment variable
        config_var = 'FLASK_CONFIG'
        config_file = os.environ.get(config_var)
        
        if config_file:
            self._load_from_file(config_var, config_file)
        else:
            self._load_from_environment()
    
    def _set_defaults(self):
        """Set default configuration values."""
        self.app.config.update(
            DEBUG=False,
            TESTING=False,
            SECRET_KEY=None,  # Must be set in production
            JSON_SORT_KEYS=False,
            JSONIFY_PRETTYPRINT_REGULAR=True
        )
    
    def _load_from_file(self, config_var, config_file):
        """Load configuration from file specified by environment variable."""
        try:
            # Verify file exists and is readable
            config_path = Path(config_file)
            if not config_path.exists():
                raise FileNotFoundError(f"Configuration file not found: {config_file}")
            
            if not config_path.is_file():
                raise ValueError(f"Configuration path is not a file: {config_file}")
            
            # Load configuration using Flask's built-in method
            self.app.config.from_envvar(config_var)
            print(f"✓ Configuration loaded from file: {config_file}")
            
        except Exception as e:
            print(f"✗ Failed to load configuration from {config_file}: {e}")
            print("Falling back to environment variables...")
            self._load_from_environment()
    
    def _load_from_environment(self):
        """Load configuration directly from environment variables."""
        env_mapping = {
            'FLASK_DEBUG': ('DEBUG', bool),
            'FLASK_SECRET_KEY': ('SECRET_KEY', str),
            'DATABASE_URL': ('DATABASE_URL', str),
            'REDIS_URL': ('REDIS_URL', str),
            'MAIL_SERVER': ('MAIL_SERVER', str),
            'MAIL_PORT': ('MAIL_PORT', int),
        }
        
        for env_key, (config_key, value_type) in env_mapping.items():
            value = os.environ.get(env_key)
            if value:
                if value_type == bool:
                    value = value.lower() in ('true', '1', 'yes', 'on')
                elif value_type == int:
                    try:
                        value = int(value)
                    except ValueError:
                        print(f"Warning: Invalid integer value for {env_key}: {value}")
                        continue
                
                self.app.config[config_key] = value
                print(f"✓ Set {config_key} from environment variable {env_key}")
 
# Usage example
def create_app():
    """Application factory with advanced configuration management."""
    app = Flask(__name__)
    
    # Initialize configuration manager
    config_manager = ConfigurationManager(app)
    
    # Validate critical configuration

        if app.config.get('DEBUG'):
            app.config['SECRET_KEY'] = 'dev-secret-key-not-for-production'
            print("Warning: Using development secret key")
        else:
            raise ValueError("SECRET_KEY must be set for production deployment")
    
    return app

Method 3: Conditional Loading with Fallbacks

Rating: ⭐⭐⭐ Best for Enterprise Applications

How it works:

This method provides a robust configuration system with multiple fallback mechanisms, environment detection, and validation. It’s designed for enterprise applications that need to work across different environments seamlessly.

Complete Implementation:

import os
import sys
from pathlib import Path
from flask import Flask
from typing import Dict, Any, Optional
 
class AdvancedConfigurationLoader:
    """Advanced configuration loader with multiple fallback strategies."""
    
    def __init__(self, app: Optional[Flask] = None):
        self.app = app
        self.config_loaded = False
        self.config_source = None
        
        if app:
            self.init_app(app)
    
    def init_app(self, app: Flask):
        """Initialize the configuration loader with a Flask app."""
        self.app = app
        self.load_configuration()
    
    def load_configuration(self):
        """Load configuration using multiple strategies with fallbacks."""
        strategies = [
            self._load_from_envvar,
            self._load_from_instance_folder,
            self._load_from_environment_variables,
            self._load_defaults
        ]
        
        for strategy in strategies:
            try:
                if strategy():
                    self.config_loaded = True
                    break
            except Exception as e:
                print(f"Configuration strategy failed: {strategy.__name__}: {e}")
                continue
        
        if not self.config_loaded:
            raise RuntimeError("Failed to load configuration from any source")
        
        self._validate_configuration()
        self._post_process_configuration()
    
    def _load_from_envvar(self) -> bool:
        """Strategy 1: Load from environment variable specified file."""
        config_var = 'FLASK_CONFIG'
        config_file = os.environ.get(config_var)
        
        if not config_file:
            return False
        
        config_path = Path(config_file)
        if not config_path.exists():
            print(f"Configuration file specified in {config_var} does not exist: {config_file}")
            return False
        
        try:
            # Use Flask's built-in method with silent=False to catch errors
            self.app.config.from_envvar(config_var, silent=False)
            self.config_source = f"environment_variable:{config_file}"
            print(f"✓ Configuration loaded from environment variable: {config_file}")
            return True
            
        except Exception as e:
            print(f"Failed to load from environment variable file: {e}")
            return False
    
    def _load_from_instance_folder(self) -> bool:
        """Strategy 2: Load from instance folder configuration files."""
        instance_configs = ['config.py', 'settings.py', 'local_config.py']
        
        for config_name in instance_configs:
            config_path = Path(self.app.instance_path) / config_name
            
            if config_path.exists():
                try:
                    # Load using from_pyfile for instance folder files
                    self.app.config.from_pyfile(config_name, silent=False)
                    self.config_source = f"instance_folder:{config_path}"
                    print(f"✓ Configuration loaded from instance folder: {config_path}")
                    return True
                    
                except Exception as e:
                    print(f"Failed to load from instance folder {config_path}: {e}")
                    continue
        
        return False
    
    def _load_from_environment_variables(self) -> bool:
        """Strategy 3: Load configuration directly from environment variables."""
        env_config = self._extract_flask_env_vars()
        
        if not env_config:
            return False
        
        self.app.config.update(env_config)
        self.config_source = "environment_variables"
        print(f"✓ Configuration loaded from environment variables: {list(env_config.keys())}")
        return True
    
    def _load_defaults(self) -> bool:
        """Strategy 4: Load default configuration as last resort."""
        default_config = {
            'DEBUG': self._detect_debug_mode(),
            'TESTING': False,
            'SECRET_KEY': self._generate_secret_key(),
            'JSON_SORT_KEYS': False,
            'JSONIFY_PRETTYPRINT_REGULAR': True,
            'MAX_CONTENT_LENGTH': 16 * 1024 * 1024,  # 16MB
        }
        
        self.app.config.update(default_config)
        self.config_source = "defaults"
        print("✓ Configuration loaded from defaults")
        return True
    
    def _extract_flask_env_vars(self) -> Dict[str, Any]:
        """Extract Flask-related environment variables."""
        env_mapping = {
            'FLASK_DEBUG': ('DEBUG', self._parse_bool),
            'FLASK_TESTING': ('TESTING', self._parse_bool),
            'FLASK_SECRET_KEY': ('SECRET_KEY', str),
            'DATABASE_URL': ('DATABASE_URL', str),
            'REDIS_URL': ('REDIS_URL', str),
            'MAIL_SERVER': ('MAIL_SERVER', str),
            'MAIL_PORT': ('MAIL_PORT', int),
            'MAIL_USE_TLS': ('MAIL_USE_TLS', self._parse_bool),
            'UPLOAD_FOLDER': ('UPLOAD_FOLDER', str),
            'MAX_CONTENT_LENGTH': ('MAX_CONTENT_LENGTH', int),
        }
        
        config = {}
        for env_key, (config_key, converter) in env_mapping.items():
            value = os.environ.get(env_key)
            if value:
                try:
                    config[config_key] = converter(value)
                except (ValueError, TypeError) as e:
                    print(f"Warning: Invalid value for {env_key}: {value} ({e})")
        
        return config
    
    def _parse_bool(self, value: str) -> bool:
        """Parse boolean values from environment variables."""
        return str(value).lower() in ('true', '1', 'yes', 'on', 'enabled')
    
    def _detect_debug_mode(self) -> bool:
        """Detect if we should enable debug mode."""
        # Check environment variable first
        if 'FLASK_DEBUG' in os.environ:
            return self._parse_bool(os.environ['FLASK_DEBUG'])
        
        # Check if running in development
        if 'pytest' in sys.modules or 'unittest' in sys.modules:
            return True
        
        # Default to False for production safety
        return False
    
    def _generate_secret_key(self) -> str:
        """Generate a secret key for development use."""
        if self.app.config.get('DEBUG'):
            return 'dev-secret-key-change-in-production'
        else:
            raise ValueError("SECRET_KEY must be explicitly set for production")
    
    def _validate_configuration(self):
        """Validate critical configuration settings."""
        required_keys = ['SECRET_KEY']
        
        for key in required_keys:
            if not self.app.config.get(key):
                raise ValueError(f"Required configuration key '{key}' is missing")
        
        # Validate SECRET_KEY strength in production
        if not self.app.config.get('DEBUG'):
            secret_key = self.app.config['SECRET_KEY']
            if len(secret_key) < 32:
                raise ValueError("SECRET_KEY must be at least 32 characters for production")
            
            if secret_key in ['dev-secret-key-change-in-production', 'development-key']:
                raise ValueError("Default development SECRET_KEY detected in production")
    
    def _post_process_configuration(self):
        """Post-process configuration after loading."""
        # Ensure instance path exists
        os.makedirs(self.app.instance_path, exist_ok=True)
        
        # Set up logging configuration
        if self.app.config.get('DEBUG'):
            import logging
            logging.basicConfig(level=logging.DEBUG)
        
        print(f"Configuration loaded successfully from: {self.config_source}")
 
# Usage with application factory
def create_app(config_override: Optional[Dict[str, Any]] = None) -> Flask:
    """Create Flask application with advanced configuration loading."""
    app = Flask(__name__)
    
    # Load configuration
    config_loader = AdvancedConfigurationLoader(app)
    
    # Apply any configuration overrides (useful for testing)
    if config_override:
        app.config.update(config_override)
    
    # Add configuration info endpoint
    @app.route('/config-info')
    def config_info():
        """Show configuration loading information."""
        return {
            'config_source': config_loader.config_source,
            'debug_mode': app.config.get('DEBUG'),
            'instance_path': app.instance_path,
            'config_keys': [key for key in app.config.keys() 
                          if not any(sensitive in key.upper() 
                                   for sensitive in ['SECRET', 'PASSWORD', 'KEY'])]
        }
    
    return app
 
# Create application instance
app = create_app()

Version-Specific Considerations

import sys
from flask import Flask
 
def version_aware_config_loading():
    """Load configuration with Python version awareness."""
    app = Flask(__name__)
    
    if sys.version_info >= (3, 8):
        # Use pathlib for Python 3.8+
        from pathlib import Path
        config_path = Path(os.environ.get('FLASK_CONFIG', 'config.py'))
        if config_path.exists():
            app.config.from_envvar('FLASK_CONFIG')
    else:
        # Fallback for older Python versions
        import os.path
        config_file = os.environ.get('FLASK_CONFIG', 'config.py')
        if os.path.exists(config_file):
            app.config.from_envvar('FLASK_CONFIG')
    
    return app

Common Problems & Solutions

Problem 1: FileNotFoundError when loading configuration

Symptoms: RuntimeError: The environment variable 'FLASK_CONFIG' is not set and as such configuration could not be loaded.

Solution:

import os

    print("Warning: FLASK_CONFIG not set or file not found, using defaults")

        SECRET_KEY='fallback-secret-key'
    )

Problem 4: Environment variable conflicts

Symptoms: Configuration values are overridden unexpectedly.

Solution:

import os
from flask import Flask
 
def load_config_with_priority():
    """Load configuration with clear priority order."""
    app = Flask(__name__)
    
    # 1. Load defaults first
    app.config.update(
        DEBUG=False,
        SECRET_KEY=None
    )
    
    # 2. Load from file if specified
    if 'FLASK_CONFIG' in os.environ:
        app.config.from_envvar('FLASK_CONFIG')
        print("Loaded from config file")
    
    # 3. Override with direct environment variables (highest priority)
    if 'FLASK_DEBUG' in os.environ:
        app.config['DEBUG'] = os.environ['FLASK_DEBUG'].lower() == 'true'
        print("DEBUG overridden by environment variable")
    
    if 'FLASK_SECRET_KEY' in os.environ:

import os
from flask import Flask
from typing import Dict, Any
 
class EnterpriseConfigManager:
    """Configuration manager for enterprise Flask applications."""
    
    def __init__(self):
        self.environments = {
            'development': 'config/dev.py',
            'testing': 'config/test.py',
            'staging': 'config/staging.py',
            'production': 'config/prod.py'
        }
    
    def create_app(self) -> Flask:
        """Create Flask app with environment-specific configuration."""
        app = Flask(__name__)
        
        # Determine environment
        environment = os.environ.get('FLASK_ENV', 'development')
        
        # Load base configuration
        app.config.from_object('config.base')
        
        # Load environment-specific configuration
        if environment in self.environments:
            config_file = self.environments[environment]
            if os.path.exists(config_file):
                # Set environment variable for from_envvar
                os.environ['FLASK_CONFIG'] = config_file
                app.config.from_envvar('FLASK_CONFIG')
        
        # Load sensitive configuration from environment
        self._load_secrets(app)
        
        # Validate configuration for environment
        self._validate_environment_config(app, environment)
        
        return app
    
    def _load_secrets(self, app: Flask):
        """Load sensitive configuration from environment variables."""
        secrets = {
            'SECRET_KEY': 'FLASK_SECRET_KEY',
            'DATABASE_PASSWORD': 'DB_PASSWORD',
            'API_KEY': 'EXTERNAL_API_KEY'
        }
        
        for config_key, env_key in secrets.items():
            value = os.environ.get(env_key)
            if value:
                app.config[config_key] = value
    
    def _validate_environment_config(self, app: Flask, environment: str):
        """Validate configuration for specific environment."""
        if environment == 'production':
            required = ['SECRET_KEY', 'DATABASE_URL']
            for key in required:
                if not app.config.get(key):
                    raise ValueError(f"Production requires {key} to be set")
            
            if app.config.get('DEBUG'):
                raise ValueError("DEBUG must be False in production")
 
# Usage

Performance-Critical Situations

# performance_config.py
from flask import Flask
import os
import json
 
class PerformanceOptimizedConfig:
    """Configuration loader optimized for performance-critical applications."""
    
    def __init__(self):
        self._config_cache = {}
        self._cache_valid = False
    
    def create_app(self) -> Flask:
        """Create Flask app with performance-optimized configuration loading."""
        app = Flask(__name__)
        
        # Use cached configuration if available
        if self._cache_valid and self._config_cache:
            app.config.update(self._config_cache)
            return app
        
        # Load and cache configuration
        self._load_and_cache_config(app)
        
        return app
    
    def _load_and_cache_config(self, app: Flask):
        """Load configuration and cache for future use."""
        # Try JSON config first (faster parsing)
        json_config = os.environ.get('FLASK_CONFIG_JSON')
        if json_config and os.path.exists(json_config):
            with open(json_config, 'r') as f:
                config_data = json.load(f)
                app.config.update(config_data)
                self._config_cache = config_data.copy()
                self._cache_valid = True
                return
        
        # Fallback to Python config file
        if 'FLASK_CONFIG' in os.environ:
            app.config.from_envvar('FLASK_CONFIG')
            # Cache the loaded configuration
            self._config_cache = {
                key: value for key, value in app.config.items()
                if not key.startswith('_')
            }
            self._cache_valid = True
 
# Usage for high-performance applications
config_loader = PerformanceOptimizedConfig()
app = config_loader.create_app()

Related Flask Configuration Operations

  • app.config.from_object(): Load configuration from a Python object or module
  • app.config.from_pyfile(): Load configuration from a Python file by filename
  • app.config.from_json(): Load configuration from a JSON file
  • app.config.from_mapping(): Load configuration from a dictionary or mapping
  • app.config.update(): Update configuration with new values

Configuration Loading Comparison

from flask import Flask

Summary

  1. Use Method 1 (Direct File Loading) for simple development setups and quick prototyping
  2. Consider Method 2 (Environment Variable Loading) for production deployments and containerized applications
  3. Implement Method 3 (Conditional Loading) for enterprise applications requiring robust configuration management
  4. Test thoroughly in your target environment before deployment
  5. Follow security best practices by never hardcoding sensitive values in configuration files

Quick Decision Guide

  • Just getting started? → Use Method 1
  • Deploying to production? → Use Method 2
  • Building enterprise software? → Use Method 3
  • Need maximum flexibility? → Combine multiple methods

Frequently Asked Questions

Q: Which method should I choose for my Flask application?

A: Choose based on your deployment complexity:

  • Simple applications: Method 1 (Direct File Loading) is sufficient
  • Production deployments: Method 2 (Environment Variable Loading) provides better security
  • Enterprise applications: Method 3 (Conditional Loading) offers maximum flexibility and robustness

Q: What happens if the environment variable is not set?

A: By default, Flask raises a RuntimeError. Use silent=True parameter to suppress errors:

app.config.from_envvar('FLASK_CONFIG', silent=True)

However, it’s better to handle this explicitly with proper fallback configuration.

Q: How do I handle sensitive configuration data?

A: Never store sensitive data in configuration files committed to version control. Instead:

  • Use environment variables for secrets
  • Use external secret management systems
  • Implement proper access controls on configuration files
# Good: Load sensitive data from environment
app.config['SECRET_KEY'] = os.environ.get('SECRET_KEY')
app.config['DATABASE_PASSWORD'] = os.environ.get('DB_PASSWORD')
 
# Bad: Hardcode sensitive data
app.config['SECRET_KEY'] = 'hardcoded-secret'  # Don't do this!

Q: Are there performance differences between configuration methods?

A: Performance differences are minimal for most applications:

  • File loading: ~0.1ms overhead per load
  • Environment variable parsing: ~0.05ms overhead
  • Conditional loading: ~0.5ms overhead due to validation

For high-performance applications, consider caching configuration after initial load.

Q: Can I combine multiple configuration loading approaches?

A: Yes, and it’s often recommended for robust applications:

def load_layered_config(app):
    """Load configuration in layers with increasing priority."""
    # 1. Load defaults
    app.config.from_object('config.defaults')
    
    # 2. Load from file if specified
    if 'FLASK_CONFIG' in os.environ:
        app.config.from_envvar('FLASK_CONFIG')
    
    # 3. Override with environment variables (highest priority)
    app.config.from_mapping({
        key: os.environ[key] for key in os.environ
        if key.startswith('FLASK_')
    })

Q: How do I debug configuration loading issues?

A: Use systematic debugging approaches:

#!/usr/bin/env python3
"""Configuration validation script for Flask applications."""
 
import os
import sys
from pathlib import Path
 
def validate_flask_config():
    """Validate Flask configuration setup."""
    issues = []
    
    # Check environment variable
    config_var = os.environ.get('FLASK_CONFIG')
    if not config_var:
        issues.append("FLASK_CONFIG environment variable not set")
    elif not Path(config_var).exists():
        issues.append(f"Configuration file does not exist: {config_var}")
    elif not Path(config_var).is_file():
        issues.append(f"Configuration path is not a file: {config_var}")
    
    # Check required environment variables
    required_env_vars = ['FLASK_SECRET_KEY']
    for var in required_env_vars:
        if not os.environ.get(var):
            issues.append(f"Required environment variable not set: {var}")
    
    # Test configuration loading
    try:
        from flask import Flask
        app = Flask(__name__)
        app.config.from_envvar('FLASK_CONFIG')
        print("✓ Configuration loads successfully")
    except Exception as e:
        issues.append(f"Configuration loading failed: {e}")
    
    # Report results
    if issues:
        print("❌ Configuration validation failed:")
        for issue in issues:
            print(f"  - {issue}")
        return False
    else:
        print("✅ Configuration validation passed")
        return True
 
if __name__ == '__main__':
    success = validate_flask_config()
    sys.exit(0 if success else 1)

from flask import Flask
 
class TestFlaskConfiguration(unittest.TestCase):
    """Test cases for Flask configuration loading."""
    
    def setUp(self):
        """Set up test environment."""
        self.app = Flask(__name__)
        self.test_config_content = '''
DEBUG = True
SECRET_KEY = 'test-secret-key'
DATABASE_URL = 'sqlite:///test.db'
'''
    
    def test_config_from_envvar_success(self):
        """Test successful configuration loading from environment variable."""
        with tempfile.NamedTemporaryFile(mode='w', suffix='.py', delete=False) as f:
            f.write(self.test_config_content)
            f.flush()
            
            # Set environment variable
            os.environ['TEST_FLASK_CONFIG'] = f.name
            
            try:
                # Load configuration
                self.app.config.from_envvar('TEST_FLASK_CONFIG')
                
                # Verify configuration was loaded
                self.assertTrue(self.app.config['DEBUG'])
                self.assertEqual(self.app.config['SECRET_KEY'], 'test-secret-key')
                self.assertEqual(self.app.config['DATABASE_URL'], 'sqlite:///test.db')
                
            finally:
                # Clean up
                os.unlink(f.name)
                if 'TEST_FLASK_CONFIG' in os.environ:
                    del os.environ['TEST_FLASK_CONFIG']
    
    def test_config_from_envvar_missing_file(self):
        """Test configuration loading with missing file."""
        os.environ['TEST_FLASK_CONFIG'] = '/nonexistent/config.py'
        
        try:
            with self.assertRaises(RuntimeError):
                self.app.config.from_envvar('TEST_FLASK_CONFIG')
        finally:
            if 'TEST_FLASK_CONFIG' in os.environ:
                del os.environ['TEST_FLASK_CONFIG']
    
    def test_config_from_envvar_silent(self):
        """Test silent configuration loading."""
        os.environ['TEST_FLASK_CONFIG'] = '/nonexistent/config.py'
        
        try:
            # Should not raise exception with silent=True

        finally:
            if 'TEST_FLASK_CONFIG' in os.environ:
